    const BASE_URL = "https://m2020m.org";

    function generateQR(elementId, url, color) {
      const encodedUrl = encodeURIComponent(url);
      const qrApiUrl =
        `https://api.qrserver.com/v1/create-qr-code/?size=240x240&data=${encodedUrl}` +
        `&color=${color}&bgcolor=ffffff&format=png&margin=10`;
      document.getElementById(elementId).src = qrApiUrl;
    }

    document.addEventListener("DOMContentLoaded", () => {
      generateQR("qr-complaints", `${BASE_URL}/report.html`, "c9a962");
      generateQR("qr-incidents", `${BASE_URL}/incident-report.html`, "DC143C");
      generateQR("qr-needlestick", `${BASE_URL}/ipc/incidents/report-needlestick.html`, "7B1E24");
    });
