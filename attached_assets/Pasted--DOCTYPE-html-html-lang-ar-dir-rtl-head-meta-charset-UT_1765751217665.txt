    <script>
        const API_URL = 'https://script.google.com/macros/s/AKfycbwktT-rBSfSixAS-gGMbm4IyEY1lNoPnLKzJNPCdaw0FqK78OqoKNy0B2kGTaRP_B_O/exec';
        
        const INCIDENT_TYPES = {
            'medication_error': 'خطأ دوائي',
            'patient_fall': 'سقوط مريض',
            'infection': 'عدوى مكتسبة',
            'diagnosis_error': 'خطأ تشخيصي',
            'procedure_error': 'خطأ إجراءات',
            'near_miss': 'كاد يحدث (Near Miss)',
            'equipment_failure': 'عطل معدات',
            'communication': 'خطأ تواصل',
            'documentation': 'خطأ توثيق',
            'other': 'أخرى'
        };
        
        const SEVERITY_LEVELS = {
            'none': { name: 'بدون ضرر', color: '#28a745' },
            'minor': { name: 'ضرر بسيط', color: '#ffc107' },
            'moderate': { name: 'ضرر متوسط', color: '#fd7e14' },
            'severe': { name: 'ضرر جسيم', color: '#dc3545' },
            'death': { name: 'وفاة', color: '#000000' }
        };
        
        const STATUS_MAP = {
            'new': 'جديد',
            'under_review': 'قيد المراجعة',
            'escalated': 'تصعيد إداري',
            'rca_required': 'يتطلب RCA',
            'rca_in_progress': 'قيد التحليل العميق',
            'in_progress': 'قيد المعالجة',
            'closed': 'مغلق'
        };
        
        let allIncidents = [];
        let filteredIncidents = [];
        
        function setDefaultDates() {
            const today = new Date();
            const firstDayOfMonth = new Date(today.getFullYear(), today.getMonth(), 1);
            
            document.getElementById('filterDateTo').value = today.toISOString().split('T')[0];
            document.getElementById('filterDateFrom').value = firstDayOfMonth.toISOString().split('T')[0];
        }
        
        async function loadReport() {
            const content = document.getElementById('reportContent');
            content.innerHTML = `
                <div class="loading">
                    <i class="fas fa-spinner fa-spin"></i>
                    <p>جاري تحميل البيانات...</p>
                </div>
            `;
            
            try {
                const params = {
                    dateFrom: document.getElementById('filterDateFrom').value,
                    dateTo: document.getElementById('filterDateTo').value,
                    department: document.getElementById('filterDepartment').value,
                    incidentType: document.getElementById('filterType').value,
                    severity: document.getElementById('filterSeverity').value,
                    status: document.getElementById('filterStatus').value,
                    limit: 500
                };
                
                const response = await fetch(API_URL, {
                    method: 'POST',
                    headers: { 'Content-Type': 'text/plain' },
                    body: JSON.stringify({ action: 'getIncidents', payload: params })
                });
                
                const result = await response.json();
                
                if (!result.ok) throw new Error(result.error || 'فشل في جلب البيانات');
                
                allIncidents = result.incidents || [];
                filteredIncidents = allIncidents;
                
                await loadIncidentDetails();
                
            } catch (err) {
                console.error('Error:', err);
                content.innerHTML = `
                    <div class="empty-state">
                        <i class="fas fa-exclamation-triangle" style="color:var(--danger);"></i>
                        <h3>خطأ في تحميل البيانات</h3>
                        <p>${err.message}</p>
                    </div>
                `;
            }
        }
        
        async function loadIncidentDetails() {
            const content = document.getElementById('reportContent');
            
            if (filteredIncidents.length === 0) {
                content.innerHTML = `
                    <div class="empty-state">
                        <i class="fas fa-inbox"></i>
                        <h3>لا توجد نتائج</h3>
                        <p>لا توجد حوادث تطابق معايير البحث المحددة</p>
                    </div>
                `;
                document.getElementById('reportStats').style.display = 'none';
                document.getElementById('reportCount').textContent = '';
                return;
            }
            
            renderStats();
            
            const detailedIncidents = [];
            
            for (const incident of filteredIncidents) {
                try {
                    const response = await fetch(API_URL, {
                        method: 'POST',
                        headers: { 'Content-Type': 'text/plain' },
                        body: JSON.stringify({ action: 'getIncidentDetails', payload: { incidentId: incident.id } })
                    });
                    const result = await response.json();
                    if (result.ok && result.incident) {
                        detailedIncidents.push({
                            ...result.incident,
                            followups: result.followups || []
                        });
                    }
                } catch (e) {
                    console.error('Error loading details for', incident.id, e);
                    detailedIncidents.push(incident);
                }
            }
            
            renderReport(detailedIncidents);
        }
        
        function renderStats() {
            const stats = document.getElementById('reportStats');
            
            const total = filteredIncidents.length;
            const closed = filteredIncidents.filter(i => i.status === 'closed').length;
            const open = total - closed;
            const rcaRequired = filteredIncidents.filter(i => i.rcaRequired).length;
            const escalated = filteredIncidents.filter(i => i.status === 'escalated').length;
            
            stats.innerHTML = `
                <div class="stat-box">
                    <div class="stat-number">${total}</div>
                    <div class="stat-label">إجمالي الحوادث</div>
                </div>
                <div class="stat-box warning">
                    <div class="stat-number">${open}</div>
                    <div class="stat-label">مفتوحة</div>
                </div>
                <div class="stat-box success">
                    <div class="stat-number">${closed}</div>
                    <div class="stat-label">مغلقة</div>
                </div>
                <div class="stat-box danger">
                    <div class="stat-number">${rcaRequired}</div>
                    <div class="stat-label">تتطلب RCA</div>
                </div>
                <div class="stat-box purple">
                    <div class="stat-number">${escalated}</div>
                    <div class="stat-label">مصعدة</div>
                </div>
            `;
            stats.style.display = 'grid';
            
            document.getElementById('reportCount').textContent = `(${total} حالة)`;
            
            const fromDate = document.getElementById('filterDateFrom').value;
            const toDate = document.getElementById('filterDateTo').value;
            if (fromDate && toDate) {
                document.getElementById('reportPeriod').textContent = `الفترة: ${fromDate} إلى ${toDate}`;
            }
        }
        
        function renderReport(incidents) {
            const content = document.getElementById('reportContent');
            
            let html = '';
            
            incidents.forEach((inc, index) => {
                const severityInfo = SEVERITY_LEVELS[inc.severity] || { name: inc.severity, color: '#6c757d' };

        function exportToExcel() {
            if (filteredIncidents.length === 0) {
                alert('لا توجد بيانات للتصدير');
                return;
            }
            
            const data = filteredIncidents.map(inc => ({
                'رقم الحادث': inc.id,
                'تاريخ الحادث': inc.date,
                'وقت الحادث': inc.time || '',
                'تاريخ التبليغ': inc.reportDate,
                'القسم': inc.department,
                'نوع الحادث': INCIDENT_TYPES[inc.type] || inc.typeName || inc.type,
                'مستوى الخطورة': SEVERITY_LEVELS[inc.severity]?.name || inc.severity,
                'الحالة': STATUS_MAP[inc.status] || inc.status,
                'الوصف': inc.description || '',
                'المسؤول': inc.assignedTo || '',
                'تصعيد إلى': inc.escalatedTo || '',
                'يتطلب RCA': inc.rcaRequired ? 'نعم' : 'لا'
            }));
            
            const ws = XLSX.utils.json_to_sheet(data);
            const wb = XLSX.utils.book_new();
            XLSX.utils.book_append_sheet(wb, ws, 'تقرير الحوادث');
            
            const fromDate = document.getElementById('filterDateFrom').value || 'all';
            const toDate = document.getElementById('filterDateTo').value || 'all';
            const filename = `تقرير_حوادث_السلامة_${fromDate}_${toDate}.xlsx`;
            
            XLSX.writeFile(wb, filename);
        }
